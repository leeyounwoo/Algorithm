visited = [False for _ in range(1000)]
graph = [[] for _ in range(1000)]


def dfs(stack):
    if not stack:
        return
    curr = stack.pop(-1)
    if not graph[curr]:
        print()
    for node in graph[curr]:
        if not visited[node]:
            visited[node] = True
            stack.append(node)
            if len(graph[curr]) == 1:

    temp = str(num)
